3.3 Taxon IDs

To link taxa with information in the extension(s), each taxon needs a unique and stable taxonID. Here we create one in the form of dataset_shortname:taxon:hash, where hash is unique code based on scientific name and kingdom. However, the hash will only remain the same as long as scientific name and kingdom remain the same. As soon as one character is changed, a new hash will be generated. Therefore, we strongly recommend to create the hash once for each taxon, and add it to the raw data file in the column taxon_id_hash. This way, the hash will remain stable even if scientific names change or are corrected in the future.

vdigest <- Vectorize(digest) # Vectorize digest function to work with vectors
taxon_id_hashes <- 
    input_data |>
  mutate(
    # Only create a hash if taxon_id_hash is missing
    taxon_id_hash = if_else(
      is.na(taxon_id_hash),
      vdigest(paste(scientific_name, kingdom), algo = "md5"),
      taxon_id_hash
      )
  ) |>
  select(taxon_id_hash, scientific_name)

taxon_id_hashes
write.csv(taxon_id_hashes, here("data", "raw", "taxon_id_hashes.csv"), na = "")

4 Taxon core

4.1 Term mapping

4.1.1 Record-level terms

Create a dataframe with unique taxa only (ignoring multiple distribution rows). Map the data to Darwin Core Taxon.

Start with record-level terms which contain metadata about the dataset (which is generally the same for all records).

taxon <-
  input_data |>
  distinct(taxon_id, .keep_all = TRUE) |>
  mutate(
    language = "en", # e.g. "en"
    license = "https://creativecommons.org/licenses/by/4.0/",
    rightsHolder = "MARE", 
    datasetID = "my_dataset_doi",
    institutionCode = "MARE",
    datasetName = "Checklist of non-native freshwater fauna in Portugal"
  )

4.1.2 Taxon-level terms

The following terms contain information about the taxon:

taxon <-

5 Create distribution extension

5.1 Term mapping

Create a dataframe with all data (including multiple distributions). Map the data to Species Distribution.

distribution <-
  input_data |>
  # pathway
  pivot_longer(
    names_to = "key", 
    values_to = "pathway", 
    starts_with("introduction_pathway"),
    values_drop_na = FALSE) |>
  filter( # keep NA value for species with no pathway provided
    !is.na(pathway) |
      (is.na(pathway) & key == "introduction_pathway_1")
      ) |> 
  # other terms
  mutate(
    taxonID = taxon_id,
    locality = location,
    countryCode = country_code,
    occurrenceStatus = occurrence_status,
    establishmentMeans = establishment_means,
    degreeOfEstablishment = degree_of_establishment,
    eventDate = case_when(
      is.na(date_first_observation) & is.na(date_last_observation) ~ NA,
      is.na(date_first_observation) ~ paste0("/", date_last_observation),
      is.na(date_last_observation) ~ paste0(date_first_observation, "/"),
      !is.na(date_first_observation) & !is.na(date_last_observation) ~ 
        paste(date_first_observation, date_last_observation, sep = "/")
    ),
    source = source,
    occurrenceRemarks = occurrence_remarks
  )

5.2 Post-processing

Only keep the Darwin Core columns and arrange by taxonID.

distribution <-
  distribution |>
  select(
    "taxonID", "locality", "countryCode", "occurrenceStatus",
    "establishmentMeans", "degreeOfEstablishment", "pathway", "eventDate",
    "source", "occurrenceRemarks"
  ) |>
  arrange(taxonID)

6 Species profile extension

In this extension we will express broad habitat characteristics of the species (e.g. isTerrestrial).

Create a dataframe with unique taxa only (ignoring multiple distribution rows). Only keep records for which terrestrial, marine and freshwater is not empty.

Map the data to Species Profile.

species_profile <-
  input_data |>
  distinct(taxon_id, .keep_all = TRUE) |> 
  filter(
    !is.na(terrestrial) |
      !is.na(marine) |
      !is.na(freshwater)
  ) |> 
  mutate(
    .keep = "none",
    taxonID = taxon_id,
    isMarine = marine,
    isFreshwater = freshwater,
    isTerrestrial = terrestrial
  ) |> 
  arrange(taxonID)

6.1 Description extension

In the description extension we want to include the native range of a species.

description <-
  input_data |> 
  # unique taxa only (ignoring multiple distribution rows)
  distinct(taxon_id, .keep_all = TRUE) |> 
  # Separate values on `|` 
  mutate(native_range = strsplit(native_range, "\\|")) |> 
  unnest(native_range) |> 
  filter(!is.na(native_range)) |> 
  mutate(
    .keep = "none",
    taxonID = taxon_id,
    description = str_trim(native_range),
    type = "native range",
    language = "en"
    ) |> 
  arrange(taxonID)
